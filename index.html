&AtServerNoContext
function getStatus()
    
    if SessionParameters.IsTesting then
        result = Undefined;

    else if SessionParameters.CurrentUser.IsAdmin then
        result = "CONNECTED";

    else 
        result = "ERROR";

    endif;

    return result;
endfunction
